3. Technical Assessment Exercise

Smart Contract Analysis

contract Token {
    address public owner;
    mapping(address => uint) balances;
    
    function withdraw() public {
        require(msg.sender == owner);
        owner.transfer(address(this).balance);
    }
    
    function transfer(address to, uint amount) public {
        balances[msg.sender] -= amount;
        balances[to] += amount;
    }
}

Code Review Quiz

What security vulnerabilities exist in this contract?

A) No balance checks in transfer function
B) Centralized owner control only
C) Integer overflow only
D) All of the above
